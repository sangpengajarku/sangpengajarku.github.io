B = 2000

[tex]gula = \frac{2}{5} \times 10 = \frac{20}{5} = 4kg \\ tepung = \frac{3}{5} \times 10 = \frac{30}{5} = 6kg \\ selisih = gula - tepung \\ = 4kg - 6kg \\ = - 2kg[/tex]
